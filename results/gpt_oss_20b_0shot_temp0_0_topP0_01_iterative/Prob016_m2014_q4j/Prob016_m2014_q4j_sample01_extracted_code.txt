// -----------------------------------------------------------------------------
//  Full‑adder helper module
// -----------------------------------------------------------------------------
module FullAdder (
    input  a,
    input  b,
    input  cin,
    output sum,
    output cout
);
    // The 2‑bit addition of a, b and cin produces a 3‑bit result.
    // The most significant bit is the carry out, the least significant bit is the sum.
    assign {cout, sum} = a + b + cin;
endmodule
