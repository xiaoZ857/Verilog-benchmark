//  -----------------------------------------
//  Input:
//      in          – single data input (positive‑edge triggered)
//      state[3:0]  – current state (one‑hot encoding)
//  Output:
//      next_state[3:0] – next state (one‑hot)
//      out           – Moore output (depends only on the current state)
//  -----------------------------------------
module TopModule (
    input                 in,        // 1‑bit data input
    input  [3:0]          state,     // current state (one‑hot)
    output reg [3:0]      next_state,// next state (one‑hot)
    output                out        // Moore output
);
