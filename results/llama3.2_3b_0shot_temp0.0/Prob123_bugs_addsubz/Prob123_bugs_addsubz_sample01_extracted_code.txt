module TopModule (
  input do_sub,
  input [7:0] a,
  input [7:0] b,
  output reg [7:0] out,
  output reg result_is_zero
);

always @(*) begin
    case (do_sub)
        0: out = a + b;
        1: out = a - b;
        default: out = a + b; // Default to addition when do_sub is not 0 or 1
    endcase

    if (~out) 
        result_is_zero = 1;
endmodule
