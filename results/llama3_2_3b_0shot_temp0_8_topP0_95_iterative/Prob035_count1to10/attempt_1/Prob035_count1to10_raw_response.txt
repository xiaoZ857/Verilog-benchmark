module TopModule (

  input reset,
  output reg [3:0] q

always_ff @(posedge clk) begin
  if (reset) begin
    count <= 1;
  } else if (count == 10) begin
    count <= 0; // Wrap around to 0 after counting to 10
  } else begin
    count <= count + 1;
  end
end
