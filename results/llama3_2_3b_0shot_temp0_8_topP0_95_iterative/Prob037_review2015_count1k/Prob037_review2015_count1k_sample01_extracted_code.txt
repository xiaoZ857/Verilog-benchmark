// Synchronous logic to increment the counter
always @(posedge clk) begin
    if (!reset) {
        count = 0; // Reset the counter to 0 on reset
    } else {
        count <= count + 1;
        
        // Ensure count wraps around after reaching 999
        if (count >= 1000) {
            count = 0; // Reset the counter to 0 on overflow
        }
    }
end
