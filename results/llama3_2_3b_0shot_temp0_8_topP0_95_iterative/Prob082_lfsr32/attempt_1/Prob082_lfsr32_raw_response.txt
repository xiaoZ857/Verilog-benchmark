  $display("Initial state: ", s);

  q = (1 << 32) - 1;

    s = (s << 1) | {q[tap_0], q[tap_2], q[tap_3]};
    q <= 1;

    s[(tap_1 + 1):31] <= s[(tap_1 + 2):32]; // Shift bits without taps.
    s[31:(tap_2 - 1)] <= {q[tap_0], s[(tap_2 + 1):32]};

  end

endmodule
