module TopModule (
  input a,
  input b,
  input c,
  input d,
  output q
);

// Create an always block to capture the transitions
always @((a, b, c, d)) begin
    // Capture the output Q value at each clock cycle
    if (a && b)
        q <= 1;
    else if (c)
        q <= 1; 
    else if (d)
        q <= 0;
    else if (!(~a & ~b & ~c & ~d))
        q <= 0; // Check for the opposite condition
end

// Add a default value for q to avoid warning
default q = 0;

endmodule
