  input wire clk,
  input wire reset,
  output reg [3:0] q
);

  if (reset) begin
    q <= 4'b0000; // Reset the counter to 0
  end else begin
    if (q < 16'b1111) begin
      q <= q + 1; // Increment the counter
    end
  end
end
