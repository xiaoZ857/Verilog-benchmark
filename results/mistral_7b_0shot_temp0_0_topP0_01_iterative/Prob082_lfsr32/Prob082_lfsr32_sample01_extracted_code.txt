        q <= 32'h1; // Reset the LFSR to the initial value 32'h1
    end else begin
        // Shift the LFSR right and XOR with tapped bits (correct tap positions)
        q <= {q[31], xor_gate(q[31]), xor_gate(q[22]), xor_gate(q[2]), xor_gate(q[0])};
    end
