 ```verilog
always @(posedge clk or negedge reset) begin
    if (!reset) begin
        out <= 32'b0; // Reset all outputs to 0
    end else begin
        always_comb begin
            for (int i = 0; i < 32; i = i + 1) begin
                if ((in[i] == 1'b1) && (out[i] == 1'b0)) begin
                    out[i] <= 1'b1; // Capture when input changes from 1 to 0
                end
            end
        end
    end
end
endmodule
```
