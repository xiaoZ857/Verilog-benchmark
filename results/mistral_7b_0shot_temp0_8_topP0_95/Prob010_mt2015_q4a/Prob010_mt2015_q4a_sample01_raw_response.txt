module TopModule (
  input wire x,
  input wire y,
  output reg z
);

always @(posedge clk) begin
  z <= ((x ^ y) & x);
end

initial begin
  #10 clk = 1; // Initial clock pulse for 10 time units (adjust as needed)
  forever #5 clk = ~clk; // Clock toggling every 5 time units (adjust as needed)
end
