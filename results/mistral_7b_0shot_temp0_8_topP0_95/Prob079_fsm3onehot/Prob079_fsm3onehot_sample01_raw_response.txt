assign C = state[2];
assign D = state[3];

always_comb begin
    // State transition logic
    if(A) begin
        next_state <= {B, A};

        next_state <= {C, B};
    end else if(C) begin
        next_state <= {A, D};

    end

    // Output logic
    out = C | (D & in);
end

endmodule
