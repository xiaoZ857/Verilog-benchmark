```verilog
module TopModule (
    input [7:0] a,
    input [7:0] b,
    input [7:0] c,
    input [7:0] d,
    output reg [7:0] min
);

assign temp_a = (a < b) ? a : b;
assign temp_b = (temp_a < c) ? temp_a : c;
assign temp_c = (temp_b < d) ? temp_b : d;

assign min = temp_c;

always @(posedge clk) begin
  if (temp_c != min) begin
    min <= temp_c;
  end
end

The comparison logic uses conditional assignment (`? :`) to find the minimum between each pair of numbers.
