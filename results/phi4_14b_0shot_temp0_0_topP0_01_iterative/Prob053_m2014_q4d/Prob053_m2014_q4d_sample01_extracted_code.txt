module TopModule (
  input clk,
  input in,
  output logic out
);

  // Intermediate wire to hold the XOR result
  wire xor_out;

  // Initialize 'out' to a known value, e.g., 0
  initial begin
    out = 1'b0;
  end

  // XOR gate implementation
  assign xor_out = in ^ out;

  // D flip-flop implementation
  always_ff @(posedge clk) begin
    out <= xor_out;
  end

endmodule
