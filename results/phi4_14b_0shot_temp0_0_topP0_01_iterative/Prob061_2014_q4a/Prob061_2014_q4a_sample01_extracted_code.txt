module TopModule (
  input clk,
  input w,
  input R,
  input E,
  input L,
  output reg Q
);

always @(posedge clk) begin
    if (L) begin
        // Load the value R into Q when L is asserted
        Q <= R;
    end else if (E) begin
        // Shift in w to Q when E is asserted and L is not asserted
        Q <= w;
    end else begin
        // Retain current state of Q otherwise
        Q <= Q;
    end
end

endmodule
