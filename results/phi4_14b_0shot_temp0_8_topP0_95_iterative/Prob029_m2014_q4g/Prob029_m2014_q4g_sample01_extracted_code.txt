module TopModule (
  input logic in1,
  input logic in2,
  input logic in3,
  output logic out
);

  // Intermediate wire to hold the result of the XNOR operation
  logic xnor_result;

  // XNOR gate connected to in1 and in2
  assign xnor_result = ~(in1 ^ in2); 

  // XOR gate with its inputs being xnor_result and in3
  assign out = xnor_result ^ in3; 

endmodule
